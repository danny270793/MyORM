    Class ModelAbstract

    Abstract base class for all database models following the Active Record pattern. Provides CRUD operations and automatic table name generation.

    Example

    class User extends Model {
        static readonly tableName = 'users';

        getId(): string | number {
            return this.id.get();
        }

        id: Column = new Column('id', 'number');
        name: Column = new Column('name', 'string');
        email: Column = new Column('email', 'string');
    }

    // Create
    const user = await User.create({ name: 'John', email: 'john@example.com' });

    // Read
    const found = await User.find('1');
    const all = await User.findAll();

    // Update
    found.name.set('Jane');
    await found.save();

    // Delete
    await found.delete();
